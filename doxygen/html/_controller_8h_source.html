510  ompl::base::State* endState,
511  ompl::base::Cost &filteringCost,
512  bool constructionMode)
513 {
514  using namespace std;
515  bool result = true;
516 
517  //HOW TO SET INITAL VALUE OF COST
518  //cost = 1 ,for time based only if time per execution is "1"
519  //cost = 0.01 , for covariance based
520 
521  // REVIEW artificial manipulation of cost for faster convergence of Dynamic Programming... can we get rid of this?
522  //double cost = 0.001;
523  double cost = 0.000; // this should be fine for Dijkstra search
524 
525  ompl::base::State *internalState = si_->allocState();
526  si_->copyState(internalState, startState);
527 
528  ompl::base::State *nominalX_K;
529 
530  ompl::base::State *tempEndState = si_->allocState();
531  si_->copyState(tempEndState, startState);
532 
533 
534  this->Evolve(internalState, k, tempEndState) ;
535  si_->copyState(internalState, tempEndState);
536 
537 
538  // if the propagated state is not valid, return false (only in construction mode)
539  if(constructionMode)
540  {
541  if(!si_->checkTrueStateValidity())

547  }
548  else if(!si_->isValid(internalState))
549  {
550  std::cout << "Failed in isValid(internalState) test!" << std::endl;
551  //si_->copyState(endState, internalState);
552  //return false;
553  result = false;
554  }
555  }
556 
557 
558  // check for deviation from the control target
559  if(k<lss_.size())
560  nominalX_K = lss_[k].getX();
561  else
562  {
563  nominalX_K = lss_[lss_.size()-1].getX();
564 
565  // quit edge controller and switch to node controller!

568  }

570  arma::colvec nomXVec = nominalX_K->as<StateType>()->getArmaData();
571  arma::colvec endStateVec = internalState->as<StateType>()->getArmaData();

573 
574  auto deviationNorm = abs(norm(deviation,2));

582  //si_->copyState(endState, internalState);

592  // 3) cost = wc * mean(trace(cov_k)) + wt * K
593  // 4) cost = wc * sum(trace(cov_k))
594 
595  // actual execution cost but only for covariance penalty (even without weight multiplication)
596  arma::mat tempCovMat = internalState->as<StateType>()->getCovariance();
597 
598  cost += arma::trace(tempCovMat); // 1)
599 // cost += arma::trace(tempCovMat); // 3) with division by the total number of time steps at the end
600 // cost += arma::trace(tempCovMat); // 4)
601 
602 
603  // no need for additional sleep for visualization during rollout
604  //if(!constructionMode)
605  // boost::this_thread::sleep(boost::posix_time::milliseconds(20));
606 
607 
608  si_->copyState(endState, internalState);
609 
610  filteringCost = ompl::base::Cost(cost);
611 
612  si_->freeState(internalState);
613  si_->freeState(tempEndState);
614 
615  //return true;
616  return result;
617 }

619 template <class SeparatedControllerType, class FilterType>
620 bool Controller<SeparatedControllerType, FilterType>::executeUpto(const int numSteps, const ompl::base::State *startState,
621  ompl::base::State* endState,
622  ompl::base::Cost &filteringCost,
623  int &stepsTaken,
624  bool constructionMode)
625 {
626  ompl::base::State *tempState = si_->allocState();
627 
628  si_->copyState(tempState, startState);
629 
630  ompl::base::State *tempEndState = si_->allocState();
631 
632  int k = 0;
633 
634  filteringCost = ompl::base::Cost(0.0);
635  ompl::base::Cost filteringCostOneStep;
636 
637  while(k < numSteps)
638  {
639 
640  bool e = executeOneStep(k, tempState,tempEndState, filteringCostOneStep, constructionMode);
641 
642  filteringCost = ompl::base::Cost(filteringCost.value() + filteringCostOneStep.value());
643 
644  k++;
645 
646  si_->copyState(tempState, tempEndState);
647 
648  si_->copyState(endState, tempEndState);
649 
650  if(!e)
651  {
652  si_->freeState(tempEndState);
653 
654  si_->freeState(tempState);
655 
656  stepsTaken = k;
657 
658  return false;
659  }
660 
661  }
662 
663  si_->freeState(tempEndState);
664 
665  si_->freeState(tempState);
666 
667  stepsTaken = k;
668 
669  return true;
670 }
